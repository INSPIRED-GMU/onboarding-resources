### Example of BufferGeometry

```json
{
	"metadata": {
		"version": 4,
		"type": "BufferGeometry",
		"generator": "BufferGeometryExporter"
	},
	"data": {
		"attributes": {
			"position": {
				"itemSize": 3,
				"type": "Float32Array",
				"array": [50,50,50,...]
			},
			"normal": {
				"itemSize": 3,
				"type": "Float32Array",
				"array": [1,0,0,...]
			},
			"uv": {
				"itemSize": 2,
				"type": "Float32Array",
				"array": [0,1,...]
			}
		},
		"boundingSphere": {
			"center": [0,0,0],
			"radius": 86.60254037844386
		}
	}
}
```
